Programa ::= Declaracoes+
Declaracoes  ::= (VariavelArray | Record | VariavelRecord | VariavelArrayDeRecord)

VariavelArray ::= Identificador ':' 'array' Inteiro '..' Inteiro 'of' TipoDeDados ';'

Record ::= 'record' Identificador 'is' ArgumentoDoRecord 'endrecord' ';'
ArgumentoDoRecord ::= Identificador ':' TipoDeDados ';'
TipoDeDados ::= 'string' | 'integer' | 'real'

VariavelRecord ::= (Identificador ':' 'record' Identificador) 
				  | Identificador ':' 'record 'Identificador '(' (Identificador | Numeros)+ ')' ';'

VariavelArrayDeRecord ::= 'array' Inteiro '..' Inteiro 'of' 'record' Identificador';'

Identificador ::= Letras (LetrasOuNumeros)+
LetrasOuNumeros ::= Letras | Numeros
Letras ::= Letra+
Letra ::= [a-z] | [A-Z]
Numeros ::= Inteiro | Real
Inteiro ::= Digito
Real ::= (Digito)+'.'(Digito)+
Digito ::= [0-9]+
