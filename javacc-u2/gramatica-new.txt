/*AnalisadorSintatico*/

Programa ::= Comando
Comando  ::= (Variavel |  Array | Record | ArrayDeRecord)+

/*SintaticoVariavel*/
Variavel ::= DeclaracaoVariavel
DeclaracaoVariavel ::= (NomeVariavel ':' ArgumentoDaVariavel) ';'
NomeVariavel ::= Letras (LetrasOuNumeros)*
ArgumentoDaVariavel ::= TipoVariavel | TipoDeDados
TipoVariavel ::= Array | Record | ArrayDeRecord
TipoDeDados ::= 'string' | 'intenger' | 'real'

/*SintaticoArray*/
Array ::= DeclaracaoArray
DeclaracaoArray ::= 'array' Inteiro '..' Inteiro 'of' TipoDeDados ';'

/*SintaticoRecord*/
Record ::= ComandoRecord
ComandoRecord ::= 'record' NomeDoRecord 'is' ArgumentoDoRecord 'endrecord' ';'
NomeDoRecord ::= Letras (LetrasOuNumeros)*
ArgumentoDoRecord::= (Variavel ':' TipoDeDados)* ';'
DeclaracaoRecord ::= Variavel ':' 'record' Variavel (ArgumentoDaDeclaracaoRecord)?
ArgumentoDaDeclaracaoRecord ::= '('Identificador+')' ';'

/*SintaticoArrayDeRecord*/
ArrayDeRecord ::= DeclaracaoArrayDeRecord
DeclaracaoArrayDeRecord ::= 'array' Inteiro '..' Inteiro 'of' 'record' Variavel ';'

/*AnalisadorLexico*/
Identificador ::= LetrasOuNumeros
String ::= '"' LetrasOuNumeros '"'
LetrasOuNumeros ::= Letras | Numeros
Letras ::= Letra+
Letra ::= [a-z] | [A-Z]
Numeros ::= Inteiro | Real
Inteiro ::= Digito
Real ::= (Digito)+'.'(Digito)+
Digito ::= [0-9]+
